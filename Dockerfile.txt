
# Stage 1: Build the React Application
FROM node:22-alpine AS build

# Build arguments for environment variables
# This allows passing the key via --build-arg during docker build
ARG API_KEY
ENV VITE_API_KEY=$API_KEY

# Set working directory
WORKDIR /app

# Copy package management files
COPY package.json package-lock.json* ./

# Update npm to the latest version to ensure compatibility and performance
RUN npm install -g npm@latest

# Install dependencies strictly (ci ensures exact versions from lockfile)
RUN npm ci

# Copy source code
COPY . .

# Build the application
# Vite will replace process.env.API_KEY with the value of VITE_API_KEY here
RUN npm run build

# Stage 2: Serve with Nginx
FROM nginx:alpine

# Remove default nginx static assets
RUN rm -rf /usr/share/nginx/html/*

# Copy built assets from builder stage
COPY --from=build /app/dist /usr/share/nginx/html

# Copy custom Nginx configuration
COPY nginx.txt /etc/nginx/conf.d/default.conf

# Security: Add a non-root user and fix permissions
# Nginx alpine image typically has a user 'nginx' (id 101).
# We ensure the pid file and cache dirs are writable by this user.
RUN touch /var/run/nginx.pid && \
    chown -R nginx:nginx /var/run/nginx.pid && \
    chown -R nginx:nginx /var/cache/nginx && \
    chown -R nginx:nginx /var/log/nginx && \
    chown -R nginx:nginx /etc/nginx/conf.d && \
    chown -R nginx:nginx /usr/share/nginx/html

# Switch to non-root user for runtime security
USER nginx

# Expose port 8080 (Cloud Run default)
EXPOSE 8080

# Start Nginx
CMD ["nginx", "-g", "daemon off;"]
