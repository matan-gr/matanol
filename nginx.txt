
server {
    listen 8080;
    server_name localhost;
    root /usr/share/nginx/html;
    index index.html;

    # Performance: Buffer sizes for high throughput
    client_body_buffer_size 10K;
    client_header_buffer_size 1k;
    client_max_body_size 8m;
    large_client_header_buffers 2 1k;

    # --- Security Headers ---
    
    # 1. Deny framing to prevent Clickjacking
    add_header X-Frame-Options "DENY" always;
    
    # 2. Prevent MIME type sniffing
    add_header X-Content-Type-Options "nosniff" always;
    
    # 3. Enable XSS Protection (Legacy browsers)
    add_header X-XSS-Protection "1; mode=block" always;
    
    # 4. Strict Transport Security (HSTS) - 1 year
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;
    
    # 5. Referrer Policy: Send origin only to cross-origin requests, full path to same-origin
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    
    # 6. Permissions Policy: Disable features not used
    add_header Permissions-Policy "camera=(), microphone=(), geolocation=(), payment=(), usb=()" always;
    
    # 7. Content Security Policy (CSP): Strictly whitelist sources
    # - frame-ancestors 'none': Prevents embedding in iframes (Clickjacking protection)
    # - object-src 'none': Disables plugins (Flash, Java, etc)
    # - base-uri 'self': Prevents base tag hijacking
    # - connect-src: Added Google Cloud APIs required for the app functionality and GenAI
    add_header Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline' https://cdn.tailwindcss.com; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com https://cdn.jsdelivr.net; img-src 'self' data: https:; font-src 'self' https://fonts.gstatic.com; connect-src 'self' https://compute.googleapis.com https://logging.googleapis.com https://generativelanguage.googleapis.com https://storage.googleapis.com https://run.googleapis.com https://sqladmin.googleapis.com https://container.googleapis.com; frame-ancestors 'none'; object-src 'none'; base-uri 'self'; form-action 'self';" always;

    # Gzip Compression
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_min_length 256;
    gzip_types
        text/plain
        text/css
        text/javascript
        application/javascript
        application/json
        application/x-javascript
        application/xml
        application/rss+xml
        image/svg+xml;

    # Cache control for static assets (Vite hashed files)
    location ~* \.(?:css|js|jpg|svg|png|gif|ico|woff|woff2)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    # SPA Routing: Fallback to index.html for client-side routing
    location / {
        try_files $uri $uri/ /index.html;
        # Never cache index.html to ensure updates are seen immediately
        add_header Cache-Control "no-store, no-cache, must-revalidate";
    }

    # Health Check Endpoint
    location /healthz {
        access_log off;
        return 200 'OK';
        add_header Content-Type text/plain;
    }

    # Error pages
    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
        root /usr/share/nginx/html;
    }
}
